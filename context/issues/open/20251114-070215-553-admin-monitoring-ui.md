### New Admin Page: User Fly Apps

**File:** `apps/app/src/components/admin/UserFlyAppsManager.tsx`

```tsx
import { useEffect, useState } from 'react';
import { supabase } from '@/lib/supabase';
import { Card, CardContent, CardHeader, CardTitle } from '@/components/ui/card';
import { Badge } from '@/components/ui/badge';
import { Button } from '@/components/ui/button';
import { Table, TableBody, TableCell, TableHead, TableHeader, TableRow } from '@/components/ui/table';

interface UserFlyApp {
  id: string;
  user_id: string;
  fly_app_name: string;
  status: 'provisioning' | 'active' | 'error' | 'deprovisioning' | 'deleted';
  region: string;
  health_status: 'healthy' | 'unhealthy' | 'unknown';
  last_health_check: string;
  created_at: string;
  monthly_cost_estimate_usd: number;
  user_email?: string;
}

export function UserFlyAppsManager() {
  const [apps, setApps] = useState<UserFlyApp[]>([]);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    loadApps();
    // Refresh every 30 seconds
    const interval = setInterval(loadApps, 30000);
    return () => clearInterval(interval);
  }, []);

  async function loadApps() {
    const { data, error } = await supabase
      .from('user_fly_apps')
      .select(`
        *,
        user_profiles!inner(email)
      `)
      .is('deleted_at', null)
      .order('created_at', { ascending: false });

    if (!error && data) {
      setApps(data.map(app => ({
        ...app,
        user_email: app.user_profiles?.email
      })));
    }
    setLoading(false);
  }

  async function handleProvision(userId: string) {
    const { error } = await supabase.functions.invoke('provision-user-fly-app', {
      body: { user_id: userId, tier: 'pro' }
    });

    if (error) {
      alert('Failed to provision: ' + error.message);
    } else {
      alert('Provisioning started');
      loadApps();
    }
  }

  async function handleDeprovision(userId: string) {
    if (!confirm('Are you sure you want to deprovision this user\'s Fly app?')) {
      return;
    }

    const { error } = await supabase.functions.invoke('deprovision-user-fly-app', {
      body: { user_id: userId, reason: 'admin_manual' }
    });

    if (error) {
      alert('Failed to deprovision: ' + error.message);
    } else {
      alert('Deprovisioning started');
      loadApps();
    }
  }

  function getStatusBadge(status: string) {
    const colors = {
      provisioning: 'bg-blue-500',
      active: 'bg-green-500',
      error: 'bg-red-500',
      deprovisioning: 'bg-yellow-500',
      deleted: 'bg-gray-500',
    };
    return <Badge className={colors[status] || 'bg-gray-500'}>{status}</Badge>;
  }

  function getHealthBadge(health: string) {
    const colors = {
      healthy: 'bg-green-500',
      unhealthy: 'bg-red-500',
      unknown: 'bg-gray-500',
    };
    return <Badge className={colors[health] || 'bg-gray-500'}>{health}</Badge>;
  }

  if (loading) return <div>Loading...</div>;

  return (
    <Card>
      <CardHeader>
        <CardTitle>User Fly Apps</CardTitle>
      </CardHeader>
      <CardContent>
        <div className="mb-4">
          <div className="grid grid-cols-4 gap-4">
            <Card>
              <CardContent className="pt-6">
                <div className="text-2xl font-bold">{apps.length}</div>
                <div className="text-sm text-muted-foreground">Total Apps</div>
              </CardContent>
            </Card>
            <Card>
              <CardContent className="pt-6">
                <div className="text-2xl font-bold text-green-600">
                  {apps.filter(a => a.status === 'active').length}
                </div>
                <div className="text-sm text-muted-foreground">Active</div>
              </CardContent>
            </Card>
            <Card>
              <CardContent className="pt-6">
                <div className="text-2xl font-bold text-red-600">
                  {apps.filter(a => a.status === 'error').length}
                </div>
                <div className="text-sm text-muted-foreground">Errors</div>
              </CardContent>
            </Card>
            <Card>
              <CardContent className="pt-6">
                <div className="text-2xl font-bold">
                  ${apps.reduce((sum, a) => sum + (a.monthly_cost_estimate_usd || 0), 0).toFixed(2)}
                </div>
                <div className="text-sm text-muted-foreground">Est. Monthly Cost</div>
              </CardContent>
            </Card>
          </div>
        </div>

        <Table>
          <TableHeader>
            <TableRow>
              <TableHead>User Email</TableHead>
              <TableHead>App Name</TableHead>
              <TableHead>Status</TableHead>
              <TableHead>Health</TableHead>
              <TableHead>Region</TableHead>
              <TableHead>Cost/Mo</TableHead>
              <TableHead>Created</TableHead>
              <TableHead>Actions</TableHead>
            </TableRow>
          </TableHeader>
          <TableBody>
            {apps.map(app => (
              <TableRow key={app.id}>
                <TableCell>{app.user_email}</TableCell>
                <TableCell>
                  <a
                    href={`https://fly.io/apps/${app.fly_app_name}`}
                    target="_blank"
                    rel="noopener noreferrer"
                    className="text-blue-600 hover:underline"
                  >
                    {app.fly_app_name}
                  </a>
                </TableCell>
                <TableCell>{getStatusBadge(app.status)}</TableCell>
                <TableCell>{getHealthBadge(app.health_status)}</TableCell>
                <TableCell>{app.region}</TableCell>
                <TableCell>${app.monthly_cost_estimate_usd?.toFixed(2) || '0.00'}</TableCell>
                <TableCell>{new Date(app.created_at).toLocaleDateString()}</TableCell>
                <TableCell>
                  <div className="flex gap-2">
                    <Button
                      size="sm"
                      variant="outline"
                      onClick={() => window.open(`https://${app.fly_app_name}.fly.dev/health`, '_blank')}
                    >
                      Health
                    </Button>
                    {app.status === 'active' && (
                      <Button
                        size="sm"
                        variant="destructive"
                        onClick={() => handleDeprovision(app.user_id)}
                      >
                        Deprovision
                      </Button>
                    )}
                    {app.status === 'error' && (
                      <Button
                        size="sm"
                        variant="default"
                        onClick={() => handleProvision(app.user_id)}
                      >
                        Retry
                      </Button>
                    )}
                  </div>
                </TableCell>
              </TableRow>
            ))}
          </TableBody>
        </Table>
      </CardContent>
    </Card>
  );
}
```

### Event Log Viewer

**File:** `apps/app/src/components/admin/FlyAppEventLog.tsx`

```tsx
import { useEffect, useState } from 'react';
import { supabase } from '@/lib/supabase';
import { Card, CardContent, CardHeader, CardTitle } from '@/components/ui/card';
import { Badge } from '@/components/ui/badge';

export function FlyAppEventLog({ appId }: { appId: string }) {
  const [events, setEvents] = useState([]);

  useEffect(() => {
    loadEvents();
  }, [appId]);

  async function loadEvents() {
    const { data } = await supabase
      .from('user_fly_app_events')
      .select('*')
      .eq('fly_app_id', appId)
      .order('created_at', { ascending: false })
      .limit(50);

    if (data) setEvents(data);
  }

  return (
    <Card>
      <CardHeader>
        <CardTitle>Event Log</CardTitle>
      </CardHeader>
      <CardContent>
        <div className="space-y-2">
          {events.map(event => (
            <div key={event.id} className="border-b pb-2">
              <div className="flex justify-between items-start">
                <div>
                  <Badge>{event.event_type}</Badge>
                  <span className="ml-2 text-sm">{event.status}</span>
                </div>
                <span className="text-xs text-muted-foreground">
                  {new Date(event.created_at).toLocaleString()}
                </span>
              </div>
              {event.error_details && (
                <div className="mt-1 text-sm text-red-600">{event.error_details}</div>
              )}
            </div>
          ))}
        </div>
      </CardContent>
    </Card>
  );
}
```

### Health Check Cron Job

Create edge function to periodically check health:

**File:** `supabase/functions/check-fly-apps-health/index.ts`

```typescript
import { serve } from "https://deno.land/std@0.208.0/http/server.ts";
import { createClient } from "https://esm.sh/@supabase/supabase-js@2";
import { checkFlyAppHealth } from "../_shared/flyClient.ts";

serve(async (req) => {
  const supabase = createClient(
    Deno.env.get("SUPABASE_URL")!,
    Deno.env.get("SUPABASE_SERVICE_ROLE_KEY")!
  );

  // Get all active apps
  const { data: apps } = await supabase
    .from("user_fly_apps")
    .select("*")
    .eq("status", "active")
    .is("deleted_at", null);

  if (!apps) {
    return new Response(JSON.stringify({ message: "No apps to check" }), {
      status: 200,
    });
  }

  const results = [];

  for (const app of apps) {
    const healthResult = await checkFlyAppHealth(app.fly_app_name);

    await supabase
      .from("user_fly_apps")
      .update({
        last_health_check: new Date().toISOString(),
        health_status: healthResult.healthy ? "healthy" : "unhealthy",
        error_message: healthResult.error || null,
      })
      .eq("id", app.id);

    if (!healthResult.healthy) {
      await supabase.from("user_fly_app_events").insert({
        fly_app_id: app.id,
        user_id: app.user_id,
        event_type: "health_check_failed",
        status: "unhealthy",
        error_details: healthResult.error,
      });
    }

    results.push({ app: app.fly_app_name, healthy: healthResult.healthy });
  }

  return new Response(JSON.stringify({ results }), { status: 200 });
});
```
